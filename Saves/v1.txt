    function drawDots() {
      canvas.width  = window.innerWidth;
      canvas.height = window.innerHeight;
      ctx.clearRect(0, 0, canvas.width, canvas.height);
 
      const cx = canvas.width  / 2;
      const cy = canvas.height / 2;
      const cols = Math.ceil(canvas.width  / SPACING);
      const rows = Math.ceil(canvas.height / SPACING);
 
      for (let r = 0; r <= rows; r++) {
        for (let c = 0; c <= cols; c++) {
          const x = c * SPACING;
          const y = r * SPACING;
          const dx = x - cx, dy = y - cy;
          const dist = Math.sqrt(dx*dx + dy*dy);
          const maxDist = Math.sqrt(cx*cx + cy*cy);
 
          // fade: strong near edges, invisible near center (±200px band)
          const centerDead = 220;
          if (dist < centerDead) continue;
 
          const t = Math.min((dist - centerDead) / (maxDist - centerDead), 1);
          const alpha = t * 0.45;

          ctx.fillStyle = isDark ? `rgba(255,255,255,${alpha})` : `rgba(0,0,0,${alpha})`;

    function formatText(text) {
      // Escapa HTML básico ANTES de qualquer coisa
      const esc = s => s.replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;');

      // Aplica inline: **texto** → pill colorido se for título de item, senão strong
      function inlineFormat(line, isPillContext) {
        return line.replace(/\*\*(.+?)\*\*/g, (_, inner) => {
          if (isPillContext) {
            return `<span class="pill-term">${esc(inner)}</span>`;
          }
          return `<strong>${esc(inner)}</strong>`;
        }).replace(/(?<!\*)\*(?!\*)(.+?)(?<!\*)\*(?!\*)/g, (_, inner) => `<em>${esc(inner)}</em>`);
      }

      const blocks = text.split(/\n{2,}/);
      const parts = [];

      blocks.forEach(block => {
        const lines = block.split('\n').filter(l => l.trim() !== '');
        if (!lines.length) return;

        // Detecta se o bloco é uma lista numerada (1. / 2. …)
        const isNumbered = lines.every(l => /^\d+\.\s/.test(l.trim()));
        // Detecta lista com hífen/bullet
        const isBullet = lines.every(l => /^[-•]\s/.test(l.trim()));

        if (isNumbered || isBullet) {
          const items = lines.map(l => {
            const match = isNumbered
              ? l.trim().match(/^(\d+)\.\s(.*)$/)
              : l.trim().match(/^[-•]\s(.*)$/);

            const content = isNumbered ? match[2] : match[1];
            const num     = isNumbered ? match[1] : null;

            // Separa "**Título**: resto" em pill + texto
            const tipMatch = content.match(/^\*\*(.+?)\*\*:\s*(.*)/s);
            let inner = '';
            if (tipMatch) {
              inner = `<span class="pill-term">${esc(tipMatch[1])}</span> ${esc(tipMatch[2])}`;
            } else {
              inner = inlineFormat(esc(content), false);
            }

            const marker = num
              ? `<span class="msg-num">${num}</span>`
              : `<span class="msg-bullet"></span>`;

            return `<div class="msg-item">${marker}<span>${inner}</span></div>`;
          });

          parts.push(`<div class="msg-list">${items.join('')}</div>`);

        } else {
          // Parágrafo normal — une linhas com espaço
          const joined = lines.map(l => inlineFormat(esc(l), false)).join(' ');
          parts.push(`<p>${joined}</p>`);
        }
      });

      return parts.join('');
    }

    function playNotifSound() {
      try {
        const AC = window.AudioContext || window.webkitAudioContext;
        if (!AC) return;
        const ac = new AC(), osc = ac.createOscillator(), gain = ac.createGain();
        osc.connect(gain); gain.connect(ac.destination);
        osc.type = 'sine';
        osc.frequency.setValueAtTime(880, ac.currentTime);
        osc.frequency.exponentialRampToValueAtTime(660, ac.currentTime + 0.12);
        gain.gain.setValueAtTime(0.07, ac.currentTime);
        gain.gain.exponentialRampToValueAtTime(0.001, ac.currentTime + 0.25);
        osc.start(ac.currentTime); osc.stop(ac.currentTime + 0.25);
      } catch(e) {}
    }

    const MOOD_EMOJI = ['', '😟', '😔', '😐', '🙂', '😄'];

    function openRegistro() {
      regData = { before: null, screen: null, sleep: null, after: null };
      regStep = 0;
      document.querySelectorAll('.mood-opt, .opt-pill').forEach(el => el.classList.remove('selected'));
      showRegStep(0);
      document.getElementById('modal-registro').classList.add('open');
    }
    function closeRegistro() {
      document.getElementById('modal-registro').classList.remove('open');
    }
    function showRegStep(step) {
      document.querySelectorAll('.reg-step').forEach((el, i) => el.classList.toggle('active', i === step));
      for (let i = 0; i < 4; i++) {
        document.getElementById('rd' + i).classList.toggle('done', i <= step);
      }
      const backBtn = document.getElementById('reg-back');
      backBtn.style.display = step > 0 ? '' : 'none';
      const nextBtn = document.getElementById('reg-next');
      nextBtn.textContent = step === 3 ? 'Salvar ✓' : 'Próximo →';
      nextBtn.disabled = regData[REG_KEYS[step]] === null;
    }
    function selectMood(key, val, el) {
      el.closest('.mood-row').querySelectorAll('.mood-opt').forEach(e => e.classList.remove('selected'));
      el.classList.add('selected');
      regData[key] = val;
      document.getElementById('reg-next').disabled = false;
    }
    function selectOpt(key, val, el) {
      el.closest('.opt-row').querySelectorAll('.opt-pill').forEach(e => e.classList.remove('selected'));
      el.classList.add('selected');
      regData[key] = val;
      document.getElementById('reg-next').disabled = false;
    }
    function regBack() { if (regStep > 0) { regStep--; showRegStep(regStep); } }
    function regNext() {
      if (regStep < 3) { regStep++; showRegStep(regStep); }
      else { saveRegistro(); }
    }
    function saveRegistro() {
      const entry = {
        date: new Date().toISOString().slice(0, 10),
        before: regData.before, screen: regData.screen,
        sleep: regData.sleep,   after: regData.after
      };
      let registros = JSON.parse(localStorage.getItem('sinus_registros') || '[]');
      registros = registros.filter(r => r.date !== entry.date);
      registros.push(entry);
      localStorage.setItem('sinus_registros', JSON.stringify(registros));
      markCheckinDone();
      closeRegistro();
      const diff = entry.after - entry.before;
      const diffTxt = diff > 0 ? 'melhorou' : diff < 0 ? 'piorou' : 'ficou igual';
      const msg = `Acabei de fazer meu registro de hoje:
- Humor agora: ${MOOD_EMOJI[entry.before]}
- Tempo de tela: ${entry.screen}
- Sono: ${entry.sleep}
- Humor depois das redes: ${MOOD_EMOJI[entry.after]}
- Meu humor ${diffTxt} depois de usar redes.`;
      sendMessage(msg);
    }

    /* ── RELATÓRIO ── */
    function openRelatorio() {
      renderRelatorio();
      document.getElementById('modal-relatorio').classList.add('open');
    }
    function closeRelatorio() {
      document.getElementById('modal-relatorio').classList.remove('open');
    }
    function renderRelatorio() {
      const DAYS_PT   = ['Dom','Seg','Ter','Qua','Qui','Sex','Sáb'];
      const registros = JSON.parse(localStorage.getItem('sinus_registros') || '[]');
      const content   = document.getElementById('report-content');

      if (registros.length === 0) {
        content.innerHTML = '<div class="report-empty">Nenhum registro ainda.<br>Use o botão 📝 para começar a registrar seu humor diário!</div>';
        return;
      }

      const today = new Date(); today.setHours(0, 0, 0, 0);
      const week  = [];
      for (let i = 6; i >= 0; i--) {
        const d = new Date(today); d.setDate(d.getDate() - i);
        const key = d.toISOString().slice(0, 10);
        week.push({ date: d, key, reg: registros.find(r => r.date === key) || null, isToday: i === 0 });
      }

      const weekHTML = week.map(w => {
        const emoji = w.reg ? MOOD_EMOJI[w.reg.before] : '·';
        return `<div class="report-day${w.isToday ? ' today' : ''}">
          <span class="report-day-emoji">${emoji}</span>
          <span class="report-day-lbl">${DAYS_PT[w.date.getDay()]}</span>
          <span class="report-day-num">${w.date.getDate()}</span>
        </div>`;
      }).join('');

      const filled   = week.filter(w => w.reg);
      const avgB     = filled.length ? Math.round(filled.reduce((s, w) => s + w.reg.before, 0) / filled.length) : 0;
      const avgA     = filled.length ? Math.round(filled.reduce((s, w) => s + w.reg.after,  0) / filled.length) : 0;
      const improved = filled.filter(w => w.reg.after > w.reg.before).length;
      const worsened = filled.filter(w => w.reg.after < w.reg.before).length;
      const trendEmoji = improved > worsened ? '📈' : worsened > improved ? '📉' : '➡️';
      const trendTxt   = improved > worsened ? 'Humor melhorou depois das redes na maioria dos dias'
                       : worsened > improved ? 'Humor piorou depois das redes na maioria dos dias'
                       : 'Humor ficou estável depois das redes';

      content.innerHTML = `
        <div class="report-week">${weekHTML}</div>
        <div class="report-stats">
          <div class="report-stat">
            <span class="report-stat-val">${filled.length}<small style="font-size:13px;font-family:'Outfit'">/7</small></span>
            <span class="report-stat-lbl">dias registrados</span>
          </div>
          <div class="report-stat">
            <span class="report-stat-val">${MOOD_EMOJI[avgB] || '—'}</span>
            <span class="report-stat-lbl">humor médio do dia</span>
          </div>
          <div class="report-stat">
            <span class="report-stat-val">${MOOD_EMOJI[avgA] || '—'}</span>
            <span class="report-stat-lbl">humor pós-redes</span>
          </div>
        </div>
        ${filled.length > 0 ? `<div class="report-trend">${trendEmoji} ${trendTxt}</div>` : ''}
      `;
    }

    /* ── fechar modais clicando no overlay ── */
    document.getElementById('modal-registro').addEventListener('click', function(e) {
      if (e.target === this) closeRegistro();
    });
    document.getElementById('modal-relatorio').addEventListener('click', function(e) {
      if (e.target === this) closeRelatorio();
    });
